From TerribleBuffTracker/scripts/install.bat:
```bat
@echo off
set "SOURCE=%~dp0..\"
set "DEST=%PROGRAMFILES(x86)%\World of Warcraft\_retail_\Interface\AddOns\TerribleBuffTracker"
if not exist "%DEST%" mkdir "%DEST%"
echo Copying TerribleBuffTracker to WoW retail addons folder...
copy /Y "%SOURCE%TerribleBuffTracker.toc" "%DEST%\"
copy /Y "%SOURCE%Core.lua" "%DEST%\"
...
echo Done! /reload in WoW to load the addon.
```

From TerribleBuffTracker/scripts/release.bat:
```bat
@echo off
setlocal enabledelayedexpansion
if "%~1"=="" ( echo Usage: release.bat ^<version^> ... exit /b 1 )
set "VERSION=%~1"
set "TAG=v%VERSION%"
git -C "%SOURCE%" tag -a "%TAG%" -m "Release %VERSION%"
git -C "%SOURCE%" push origin main "%TAG%"
```
